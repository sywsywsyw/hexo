```javascript
// 转为http;
var targetProtocol = "http:";
// 转为https;
var targetProtocol = "https:";
if (window.location.protocol != targetProtocol)
  window.location.href = targetProtocol +
  window.location.href.substring(window.location.protocol.length);
```
