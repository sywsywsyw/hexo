#### el.onclick=(function(){})()

``` javascript
          // el.onclick=function(){
          //      // alert(1)
          //      this.className="da selected";
          // }
el.onclick=(function(){
     return function(e){  //浏览器解析是一定会传输一个实参 可以先给函数定义一个形参
          var kaiguan=true;
          if(e.shiftKey&&e.altKey&&e.ctrlKey){
               this.className=kaiguan?"da selected":"da";
               kaiguan=!kaiguan;
          }else{
               this.className="da"
          }
     }
          })();              // 函数的自调用
```
