#### 判断小程序环境
```js
  judgeWxApp() {
    var ua = navigator.userAgent.toLowerCase();
    if (ua.match(/miniProgram/i) == "miniprogram") {
      return true;
    }
    return false;
  }
  ```
#### 判断微信浏览器环境
```js
  judgeWxWeb() {
    var ua = navigator.userAgent.toLowerCase();
    if (ua.match(/MicroMessenger/i) == "micromessenger") {
      return true;
    }
    return false;
  }
 ```
